2. Create tests/test_byol.py:
   ```python
   import torch
   import pytest
   from core.config import TrainConfig, BYOLConfig
   from methods.byol.module import BYOLModule

   def make_toy_cfg():
       return TrainConfig(
           method='byol',
           backbone='resnet18',
           pretrained=False,
           max_epochs=1,
           batch_size=4,
           lr=0.001,
           byol=BYOLConfig(),
       )

   def make_toy_batch(B=4, C=3, H=32, W=32):
       v1 = torch.randn(B, C, H, W)
       v2 = torch.randn(B, C, H, W)
       labels = torch.zeros(B, dtype=torch.long)
       return (v1, v2), labels

   def test_byol_target_zero_grad():
       """Target branch parameters must receive zero gradient during training_step."""
       cfg = make_toy_cfg()
       module = BYOLModule(cfg)
       module.train()
       batch = make_toy_batch()
       loss = module.training_step(batch, 0)
       loss.backward()
       # Target params must have no gradient
       for name, p in module.backbone_ema.named_parameters():
           assert p.grad is None or p.grad.abs().max() == 0, \
               f"backbone_ema.{name} has non-zero gradient — stop-gradient broken"
       for name, p in module.projector_ema.named_parameters():
           assert p.grad is None or p.grad.abs().max() == 0, \
               f"projector_ema.{name} has non-zero gradient — stop-gradient broken"

   def test_byol_online_params_have_grad():
       """Online params must receive gradient so learning can happen."""
       cfg = make_toy_cfg()
       module = BYOLModule(cfg)
       module.train()
       batch = make_toy_batch()
       loss = module.training_step(batch, 0)
       loss.backward()
       online_params_with_grad = [
           p for p in list(module.backbone.parameters()) +
                       list(module.projector.parameters()) +
                       list(module.predictor.parameters())
           if p.grad is not None and p.grad.abs().max() > 0
       ]
       assert len(online_params_with_grad) > 0, "No online params have gradient"

   def test_byol_instantiation():
       cfg = make_toy_cfg()
       module = BYOLModule(cfg)
       assert hasattr(module, 'backbone_ema')
       assert hasattr(module, 'projector_ema')
       assert hasattr(module, 'predictor')

   def test_byol_training_step_returns_finite_loss():
       cfg = make_toy_cfg()
       module = BYOLModule(cfg)
       module.train()
       batch = make_toy_batch()
       loss = module.training_step(batch, 0)
       assert torch.isfinite(loss), f"Loss is not finite: {loss}"

   def test_byol_target_frozen_at_init():
       cfg = make_toy_cfg()
       module = BYOLModule(cfg)
       for p in module.backbone_ema.parameters():
           assert not p.requires_grad
       for p in module.projector_ema.parameters():
           assert not p.requires_grad
   ```
