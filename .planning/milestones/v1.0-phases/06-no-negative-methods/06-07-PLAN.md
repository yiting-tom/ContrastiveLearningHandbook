```yaml
# BYOL (Grill et al., NeurIPS 2020) -- ResNet-18 tutorial config
# Bootstrap Your Own Latent: A New Approach to Self-Supervised Learning
#
# No-negative SSL via online/target bootstrap. The predictor asymmetry
# (online branch only) combined with EMA prevents collapse.
#
# Original paper uses ResNet-50 with batch_size=4096 and LARS.
# This tutorial config uses ResNet-18 with batch_size=256 and AdamW.
# Performance will be lower but the method dynamics are preserved.
#
# Key: EMA momentum schedule (0.996->1.0) matters more than the base value.
# Constant momentum (not scheduled) leads to ~1% worse top-1.

method: byol
backbone: resnet18
pretrained: false

max_epochs: 200
warmup_epochs: 10
batch_size: 256
lr: 3e-4
weight_decay: 1e-6
optimizer: adamw
scheduler: warmup_cosine
n_views: 2
data_dir: data
num_workers: 4

byol:
  base_momentum: 0.996
  end_momentum: 1.0
```

# SimSiam (Chen & He, CVPR 2021) -- ResNet-18 tutorial config
# Exploring Simple Siamese Representation Learning
#
# No-negative SSL with shared encoder and stop-gradient.
# The .detach() on projection outputs is load-bearing — removing it
# causes loss to collapse to -1.0 within 2 epochs.
#
# Original paper uses ResNet-50 with batch_size=512 and SGD.
# This tutorial config uses ResNet-18 — dynamics preserved.
#
# Note: SimSiam is sensitive to learning rate. Too high → collapse.
# The paper uses lr = 0.05 * batch_size / 256 (linear scaling rule).

method: simsiam
backbone: resnet18
pretrained: false

lr: 0.05

# Barlow Twins (Zbontar et al., ICML 2021) -- ResNet-18 tutorial config
# Barlow Twins: Self-Supervised Learning via Redundancy Reduction
#
# No-negative SSL via cross-correlation matrix redundancy reduction.
# The 8192-dim projector is critical — low-dim projectors (128, 256)
# produce significantly worse representations.
#
# Original paper uses ResNet-50 with batch_size=2048 and LARS.
# This tutorial config uses ResNet-18 with batch_size=256.
# With small batch sizes, the cross-correlation statistics are noisier
# but the method still learns useful representations.
#
# lambda_coeff=5e-3 balances invariance (diagonal) vs redundancy reduction
# (off-diagonal). Higher lambda → more redundancy reduction, less invariance.

method: barlow_twins
backbone: resnet18
pretrained: false

max_epochs: 200
warmup_epochs: 10
batch_size: 256
lr: 3e-4
weight_decay: 1e-6
optimizer: adamw
scheduler: warmup_cosine
n_views: 2
data_dir: data
num_workers: 4

barlow_twins:
  lambda_coeff: 5e-3
  projection_dim: 8192
```

def test_byol_ema_momentum_schedule():
    """EMA momentum at step 0 is base_momentum; at total_steps is end_momentum."""
    from core.ema import EMAUpdater
    base, end, total = 0.996, 1.0, 1000
    ema = EMAUpdater(base_momentum=base, end_momentum=end, total_steps=total)
    assert abs(ema.momentum - base) < 1e-6, f"At step 0, momentum should be {base}"
    # Advance to final step
    for _ in range(total):
        ema._step += 1
    assert abs(ema.momentum - end) < 1e-6, f"At step {total}, momentum should be {end}"
```
