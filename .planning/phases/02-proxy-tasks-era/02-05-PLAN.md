    ```yaml
    # Instance Discrimination (Wu et al., CVPR 2018)
    # Non-parametric memory bank + NCE loss with fixed Z.
    #
    # Key hyperparameters:
    #   temperature: 0.07 (standard for NCE-based methods)
    #   n_negatives: 4096 (sampled from memory bank each step)
    #   n_views: 1 (single augmented view; bank provides comparison target)
    #
    # Note: Uses weak augmentation (era-1 defaults, no Gaussian blur).
    # Note: Memory bank features become stale as encoder trains.
    #        See MoCo (Phase 4) for the queue-based solution.

    method: instance_discrimination
    backbone: resnet18
    pretrained: false

    max_epochs: 200
    warmup_epochs: 10
    batch_size: 128
    lr: 0.03
    weight_decay: 1e-4
    optimizer: sgd
    n_views: 1
    data_dir: data
    num_workers: 4

    instance_discrimination:
      temperature: 0.07
      n_negatives: 4096
      projection_dim: 128
    ```

    3. Create `configs/invariant_spread_resnet18.yaml`:

    ```yaml
    # Invariant Spread (Ye et al., CVPR 2019)
    # In-batch contrastive learning without memory bank or queue.
    #
    # Key hyperparameters:
    #   temperature: 0.07
    #   n_views: 2 (two augmented views per image)
    #
    # IMPORTANT: Batch-size sensitive! Performance degrades significantly below
    # batch_size=256 because negatives come only from the current batch
    # (effective negatives = batch_size - 1). Unlike Instance Discrimination
    # (4096 negatives from bank) or MoCo (65536 from queue), this method's
    # negative count is limited by batch size.
    #
    # Note: Uses weak augmentation (era-1 defaults, no Gaussian blur).
    # Note: Direct ancestor of SimCLR (Phase 3).

    method: invariant_spread
    backbone: resnet18
    pretrained: false

    max_epochs: 200
    warmup_epochs: 10
    batch_size: 256
    lr: 0.03
    weight_decay: 1e-4
    optimizer: sgd
    n_views: 2
    data_dir: data
    num_workers: 4

    invariant_spread:
      temperature: 0.07
    ```

    ```python
        """End-to-end: load YAML config, instantiate module, train 1 epoch."""
        import yaml
        from core.config import TrainConfig
        from core.dispatcher import method_dispatcher
        from core.data import SSLDataModule, IndexedDataset, ssl_collate_with_index

        # Load and adapt config for test
        with open("configs/instance_discrimination_resnet18.yaml") as f:
            raw = yaml.safe_load(f)
        raw["data_dir"] = str(tmp_imagefolder)
        raw["max_epochs"] = 1
        raw["warmup_epochs"] = 0
        raw["batch_size"] = 4
        raw["num_workers"] = 0
        cfg = TrainConfig.model_validate(raw)

        import methods  # trigger registration

        # Setup data with IndexedDataset wrapper
    ```
